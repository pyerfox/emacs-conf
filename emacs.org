  ;; Initialize package sources

  (require 'tcl)
  (defun ecfg/tcl-eval-line-or-region (start end &optional and-go)
    (interactive "r\nP")
    (if (region-active-p)
      (call-interactively #'tcl-eval-region)
      (save-mark-and-excursion
        (beginning-of-line)
        (set-mark-command nil)
        (end-of-line)

        (pop-mark))))
